import numpy as np
import matplotlib.pyplot as plt

# Synthetic data
np.random.seed(0)
X = np.random.randn(200, 2)
true_theta = np.array([2.0, -3.5])
y = X @ true_theta + np.random.randn(200) * 0.5

def mse(theta, X, y):
    preds = X @ theta
    return 0.5 * np.mean((preds - y) ** 2)

def grad_mse(theta, X, y):
    preds = X @ theta
    return X.T @ (preds - y) / X.shape[0]

def plot_contour(theta_history):
    # Grid for contour
    w1 = np.linspace(-1, 5, 200)
    w2 = np.linspace(-6, 1, 200)
    W1, W2 = np.meshgrid(w1, w2)
    Z = np.array([mse(np.array([w1_, w2_]), X, y) for w1_, w2_ in zip(W1.ravel(), W2.ravel())])
    Z = Z.reshape(W1.shape)
    plt.contourf(W1, W2, Z, levels=30, cmap='viridis')
    plt.colorbar(label='MSE')
    # Trajectory
    hist = np.array(theta_history)
    plt.plot(hist[:,0], hist[:,1], 'r.-', label='GD path')
    plt.scatter(*true_theta, c='white', edgecolors='black', s=100, label='True θ')
    plt.xlabel(r'$\theta_1$')
    plt.ylabel(r'$\theta_2$')
    plt.legend()
    plt.show()

def train(optimizer='gd', lr=0.1, epochs=100, batch_size=None):
    theta = np.random.randn(2)
    history = [theta.copy()]
    for epoch in range(epochs):
        if optimizer == 'sgd':
            idx = np.random.randint(0, X.shape[0])
            Xb, yb = X[idx:idx+1], y[idx:idx+1]
        elif optimizer == 'mini':
            if batch_size is None: batch_size = 32
            ids = np.random.choice(X.shape[0], batch_size, replace=False)
            Xb, yb = X[ids], y[ids]
        else:  # full‑batch GD
            Xb, yb = X, y

        g = grad_mse(theta, Xb, yb)

        if optimizer == 'adam':
            # Simple Adam implementation (β1=β2=0.9, ε=1e-8)
            if epoch == 0:
                m = np.zeros_like(theta)
                v = np.zeros_like(theta)
            m = 0.9 * m + 0.1 * g
            v = 0.9 * v + 0.1 * (g ** 2)
            m_hat = m / (1 - 0.9 ** (epoch + 1))
            v_hat = v / (1 - 0.9 ** (epoch + 1))
            theta -= lr * m_hat / (np.sqrt(v_hat) + 1e-8)
        else:
            theta -= lr * g

        history.append(theta.copy())
        if epoch % 10 == 0:
            print(f'Epoch {epoch:3d} | Loss {mse(theta, X, y):.4f}')
    return theta, history

# Example run
theta_opt, hist = train(optimizer='gd', lr=0.05, epochs=200)
plot_contour(hist)
