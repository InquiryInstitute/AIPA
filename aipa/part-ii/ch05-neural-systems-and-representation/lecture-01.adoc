python
import numpy as np

def step(z): return 1 if z > 0 else 0

X = np.array([[0,0],[0,1],[1,0],[1,1]])
y_and = np.array([0,0,0,1])
w = np.zeros(2)
b = 0.0
eta = 1.0

for epoch in range(10):
    errors = 0
    for xi, yi in zip(X, y_and):
        o = step(np.dot(w, xi) + b)
        if o != yi:
            w += eta * (yi - o) * xi
            b += eta * (yi - o)
            errors += 1
    if errors == 0: break
print(w, b)
