import pandas as pd, numpy as np
train = pd.read_csv('train.csv')
live  = pd.read_csv('live.csv')

train_mean = train['income'].mean()
live_mean  = live['income'].mean()
drift = abs(live_mean - train_mean) / train_mean

threshold = 0.05   # 5 % shift triggers retrain
if drift > threshold:
    print(f'Drift detected: {drift:.2%} > {threshold:.2%}')
