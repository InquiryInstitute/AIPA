def retrieve_and_augment(query, k=5):
    q_emb = embed(query)
    chunks = vector_store.search(q_emb, top_k=k)      # → list of (text, score)
    context = "\n---\n".join([c.text for c in chunks])
    prompt = f"""You are a helpful support assistant.
Context:
{context}

User: {query}
Assistant:"""
    return prompt
